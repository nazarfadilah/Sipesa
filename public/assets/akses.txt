use PhpOffice\PhpSpreadsheet\IOFactory;
use PhpOffice\PhpSpreadsheet\Writer\Xlsx;
use App\Models\SampahTerkelola;
use App\Models\SampahDiserahkan;
use Carbon\Carbon;

public function exportFinal(Request $request)
{
    // 1. Ambil Input dari Form
    $tahun = $request->input('tahun'); // Misal: 2024
    $instansiIds = $request->input('instansi_ids'); // Array: [1, 3, 5] (Sesuai checkbox)
    
    // 2. Load Template Master
    $templatePath = storage_path('app/public/templates/template_master.xlsx');
    $spreadsheet = IOFactory::load($templatePath);

    // ==========================================
    // BAGIAN 1: MENGISI SHEET HARIAN (JULI - JUNI)
    // ==========================================
    
    // Mapping Nama Bulan ke Nama Sheet (Sesuaikan dengan nama sheet di Excel)
    $months = [
        7 => 'Juli', 8 => 'Agustus', 9 => 'September', 10 => 'Oktober', 11 => 'November', 12 => 'Desember',
        1 => 'Januari', 2 => 'Februari', 3 => 'Maret', 4 => 'April', 5 => 'Mei', 6 => 'Juni'
    ];

    foreach ($months as $monthNum => $sheetName) {
        $sheet = $spreadsheet->getSheetByName($sheetName);
        
        // Tentukan tahun (Juli-Des = Tahun T, Jan-Juni = Tahun T+1)
        $currentYear = ($monthNum >= 7) ? $tahun : $tahun + 1;
        
        // Query Data Harian Filter Instansi
        $dataHarian = SampahTerkelola::whereYear('tanggal', $currentYear)
            ->whereMonth('tanggal', $monthNum)
            ->whereHas('user', function($q) use ($instansiIds) {
                $q->whereIn('id_instansi', $instansiIds);
            })
            ->get();

        // LOGIKA MAPPING CELL (Ini yang paling PR)
        // Anda harus tahu baris awal tanggal 1 di Excel itu baris ke berapa.
        // Misal: Tanggal 1 ada di Baris 10.
        $startRow = 10; 

        foreach ($dataHarian as $data) {
            $day = Carbon::parse($data->tanggal)->day;
            $row = $startRow + ($day - 1); // Jika tgl 1 = baris 10, maka tgl 2 = baris 11

            // Tentukan Kolom berdasarkan Lokasi & Jenis
            // Contoh Logic (Anda harus cek Excel aslinya kolom ke berapa hurufnya):
            // Area Kantor (Lokasi ID 1) -> Organik (Jenis ID 1) -> Kolom C
            // Area Kantor (Lokasi ID 1) -> Anorganik (Jenis ID 2) -> Kolom D
            
            $col = null;
            if ($data->id_lokasi == 1) { // Area Kantor
                if ($data->id_jenis == 1) $col = 'C';
                if ($data->id_jenis == 2) $col = 'D';
                if ($data->id_jenis == 3) $col = 'E';
            } elseif ($data->id_lokasi == 2) { // Area Parkir
                if ($data->id_jenis == 1) $col = 'G'; // Misal kolom G
                // ... dst
            }

            if ($col) {
                // Penjumlahan (Jika ada 2 user input di hari sama, lokasi sama)
                $currentVal = $sheet->getCell($col . $row)->getValue();
                $sheet->setCellValue($col . $row, $currentVal + $data->jumlah_berat);
            }
        }
        
        // Lakukan hal yang sama untuk SampahDiserahkan...
    }

    // ==========================================
    // BAGIAN 2: MENGISI REKAP (Opsional jika Excel sudah pakai Rumus)
    // ==========================================
    // Jika Sheet Rekap di Excel sudah pakai rumus =SUM('Juli'!C10:C40),
    // MAKA ANDA TIDAK PERLU KODING BAGIAN INI! 
    // Biarkan Excel menghitung otomatis dari data harian yang sudah kita inject di atas.
    // Ini keuntungan besar metode Template Injection.

    // 3. Download File
    $writer = new Xlsx($spreadsheet);
    $fileName = 'Laporan_Neraca_' . time() . '.xlsx';
    
    header('Content-Type: application/vnd.openxmlformats-officedocument.spreadsheetml.sheet');
    header('Content-Disposition: attachment; filename="'. $fileName .'"');
    $writer->save('php://output');
    exit;
}

Karena kita tidak pakai rumus Excel, kita harus menggunakan teknik "Running Total" (Penjumlahan Berjalan) di dalam kodingan PHP.

Logikanya: Saat kita meloop data harian untuk mengisi Sheet Bulan (misal Juli), kita sekalian "menabung" (menjumlahkan) angkanya ke dalam variabel PHP. Setelah selesai loop bulan itu, tabungan angka tadi kita tulis ke Sheet Rekap.

Langkah-Langkah Koding:
Load Template: Buka file Excel kosong.

Loop 12 Bulan (Juli s.d Juni):

Query Database: Ambil data sampah di bulan tersebut (filter instansi & tahun).

Loop Data Harian:

Tulis angka berat ke Sheet Harian (misal sel C10).

HITUNG TOTAL (Coding): Tambahkan angka berat ke variabel total_bulanan.

Tulis ke Rekap Area: Ambil total_bulanan tadi, tulis ke Sheet Rekap Area di baris bulan Juli.

Tulis ke Rekap Terkelola: Tulis juga ke Sheet Rekap Sampah Terkelola.

Tabung untuk Tahunan: Tambahkan total_bulanan ke variabel total_tahunan.

Tulis Rekap Tahunan: Setelah selesai 12 bulan, tulis total_tahunan ke Sheet Rekap Neraca.

Download: Berikan file ke user.

private function getKolomExcel($lokasiId, $jenisId) {
    // Contoh Mapping manual
    if ($lokasiId == 1) { // KANTOR
        if ($jenisId == 1) return 'C'; // Organik

PHP

<?php

namespace App\Http\Controllers;

use Illuminate\Http\Request;
use PhpOffice\PhpSpreadsheet\IOFactory;
use PhpOffice\PhpSpreadsheet\Writer\Xlsx;
use App\Models\SampahTerkelola; // Pastikan Model Sesuai
use App\Models\SampahDiserahkan;
use App\Models\User;
use Carbon\Carbon;

class LaporanController extends Controller
{
    public function exportFinal(Request $request)
    {
        // 1. AMBIL INPUT DARI FORM
        $tahun = $request->input('tahun'); // Contoh: 2025
        $instansiIds = $request->input('instansi_ids'); // Array [1, 2] dari checkbox
        
        // 2. LOAD TEMPLATE
        $templatePath = storage_path('app/public/templates/template_master.xlsx');
        if (!file_exists($templatePath)) {
            return back()->with('error', 'File template tidak ditemukan!');
        }
        
        $spreadsheet = IOFactory::load($templatePath);

        // ==========================================
        // BAGIAN A: TULIS JUDUL INSTANSI (HEADER)
        // ==========================================
        $namaInstansiList = User::whereIn('id_instansi', $instansiIds)
                            ->join('instansis', 'users.id_instansi', '=', 'instansis.id') // Sesuaikan nama tabel instansi
                            ->distinct()
                            ->pluck('instansis.nama_instansi') // Sesuaikan nama kolom
                            ->toArray();

        $judulInstansi = "";
        if (count($namaInstansiList) == 1) {
            $judulInstansi = strtoupper($namaInstansiList[0]);
        } elseif (count($namaInstansiList) > 1 && count($namaInstansiList) <= 3) {
            $judulInstansi = strtoupper(implode(", ", $namaInstansiList));
        } else {
            $judulInstansi = "GABUNGAN BEBERAPA UNIT KERJA";
        }

        // Tembak Judul ke Sheet Rekap (Misal di Cell C3 - Sesuaikan dengan Excel Anda)
        // Cek dulu sheetnya ada atau tidak
        $sheetRekap = $spreadsheet->getSheetByName('Rekap Neraca Pengelolaan Sampah');
        if($sheetRekap) {
            $sheetRekap->setCellValue('C3', "Area: " . $judulInstansi);
        }

        // ==========================================
        // BAGIAN B: ISI DATA HARIAN (LOOPING BULAN)
        // ==========================================
        
        // Mapping: Angka Bulan => Nama Sheet
        $months = [
            7 => 'Juli', 8 => 'Agustus', 9 => 'September', 10 => 'Oktober', 11 => 'November', 12 => 'Desember',
            1 => 'Januari', 2 => 'Februari', 3 => 'Maret', 4 => 'April', 5 => 'Mei', 6 => 'Juni'
        ];

        // Posisi Baris Tanggal 1 (Cek Excel Anda, misal Tgl 1 ada di baris 10)
        $startRow = 10; 

        foreach ($months as $monthNum => $sheetName) {
            $sheet = $spreadsheet->getSheetByName($sheetName);
            
            if (!$sheet) continue; // Skip jika sheet tidak ketemu

            // Tentukan Tahun (Juli-Des = Tahun Pilihan, Jan-Juni = Tahun Berikutnya)
            $currentYear = ($monthNum >= 7) ? $tahun : $tahun + 1;

            // --- QUERY DATA SAMPAH TERKELOLA ---
            $dataTerkelola = SampahTerkelola::whereYear('tanggal', $currentYear)
                ->whereMonth('tanggal', $monthNum)
                ->whereHas('user', function($q) use ($instansiIds) {
                    $q->whereIn('id_instansi', $instansiIds);
                })->get();

            // --- PROSES DATA ---
            foreach ($dataTerkelola as $row) {
                // 1. Tentukan Baris (Row) berdasarkan Tanggal
                $tgl = (int)Carbon::parse($row->tanggal)->format('d');
                $barisExcel = $startRow + ($tgl - 1); 

                // 2. Tentukan Kolom (Column) berdasarkan Lokasi & Jenis
                // Fungsi ini ada di bawah (private function)
                $kolomExcel = $this->getExcelColumn($row->id_lokasi, $row->id_jenis);

                if ($kolomExcel) {
                    // Ambil nilai lama (jika ada 2 input di hari yang sama)
                    $valLama = $sheet->getCell($kolomExcel . $barisExcel)->getValue();
                    // Jumlahkan
                    $sheet->setCellValue($kolomExcel . $barisExcel, $valLama + $row->jumlah_berat);
                }
            }

            // --- QUERY DATA SAMPAH DISERAHKAN (Jika strukturnya sama, lakukan hal serupa) ---
            // ... (Copy paste logika di atas untuk SampahDiserahkan jika perlu) ...
        }

        // 3. DOWNLOAD FILE
        $writer = new Xlsx($spreadsheet);
        $fileName = 'Laporan_Neraca_Sampah_' . $tahun . '.xlsx';
        
        return response()->streamDownload(function() use ($writer) {
            $writer->save('php://output');
        }, $fileName);
    }

    // ==========================================
    // FUNGSI PENTING: PETA LOKASI KOLOM EXCEL
    // ==========================================
    private function getExcelColumn($lokasiId, $jenisId)
    {
        // MAPPING MANUAL (SESUAIKAN DENGAN EXCEL ASLI ANDA)
        // Contoh ini berdasarkan file CSV yang Anda kirim
        
        // LOKASI 1: AREA KANTOR
        if ($lokasiId == 1) { 
            if ($jenisId == 1) return 'C'; // Organik
            if ($jenisId == 2) return 'D'; // Anorganik
            if ($jenisId == 3) return 'E'; // Residu
        }

        // LOKASI 2: AREA PARKIR/TAMAN
        // Biasanya selang 1 kolom untuk "Total" (Kolom F dilewati)
        if ($lokasiId == 2) { 
            if ($jenisId == 1) return 'G'; // Organik
            if ($jenisId == 2) return 'H'; // Anorganik
            if ($jenisId == 3) return 'I'; // Residu
        }

        // LOKASI 3: RUANG TUNGGU
        if ($lokasiId == 3) { 
            if ($jenisId == 1) return 'K'; 
            if ($jenisId == 2) return 'L';
            if ($jenisId == 3) return 'M';
        }

        // LOKASI 4: TEMPAT MAKAN (Kantin)
        if ($lokasiId == 4) { 
            if ($jenisId == 1) return 'O'; 
            if ($jenisId == 2) return 'P';
            if ($jenisId == 3) return 'Q';
        }

        // LOKASI 5: SAMPAH KAPAL
        if ($lokasiId == 5) { 
            if ($jenisId == 1) return 'S'; 
            if ($jenisId == 2) return 'T';
            if ($jenisId == 3) return 'U';
        }

        return null; // Jika tidak ketemu
    }
}

PHP

<?php

namespace App\Http\Controllers;

use Illuminate\Http\Request;
use PhpOffice\PhpSpreadsheet\IOFactory;
use PhpOffice\PhpSpreadsheet\Writer\Xlsx;
use App\Models\SampahTerkelola;
use App\Models\SampahDiserahkan;
use App\Models\User;
use Carbon\Carbon;

class LaporanController extends Controller
{
    public function exportFinal(Request $request)
    {
        // 1. SETTING AWAL
        $tahun = $request->input('tahun'); 
        $instansiIds = $request->input('instansi_ids'); // Array dari checkbox
        
        // Load Template Master
        $templatePath = storage_path('app/public/templates/template_master.xlsx');
        if (!file_exists($templatePath)) {
            return back()->with('error', 'File template master belum diupload!');
        }
        
        // Load Excel
        $spreadsheet = IOFactory::load($templatePath);

        // ====================================================
        // BAGIAN 1: ISI JUDUL INSTANSI (HEADER)
        // ====================================================
        
        // Ambil nama instansi untuk dijadikan Judul
        $namaInstansi = User::whereIn('id_instansi', $instansiIds)
                        ->join('instansis', 'users.id_instansi', '=', 'instansis.id')
                        ->distinct()
                        ->pluck('instansis.nama_instansi')
                        ->toArray();

        // Logika Nama Judul
        $judulText = "SELURUH WILAYAH OPERASIONAL";
        if (count($namaInstansi) == 1) {
            $judulText = strtoupper($namaInstansi[0]);
        } elseif (count($namaInstansi) > 1 && count($namaInstansi) <= 3) {
            $judulText = strtoupper(implode(", ", $namaInstansi));
        }

        // Tulis Judul ke Sheet Rekap (Cell C3 sesuai layout)
        $sheetRekap = $spreadsheet->getSheetByName('Rekap Neraca Pengelolaan Sampah');
        if($sheetRekap) {
            $sheetRekap->setCellValue('C3', "Area : " . $judulText);
        }

        // ====================================================
        // BAGIAN 2: LOOPING PENGISIAN DATA HARIAN
        // ====================================================

        // Mapping Angka Bulan ke Nama Sheet
        $months = [
            7 => 'Juli', 8 => 'Agustus', 9 => 'September', 10 => 'Oktober', 11 => 'November', 12 => 'Desember',
            1 => 'Januari', 2 => 'Februari', 3 => 'Maret', 4 => 'April', 5 => 'Mei', 6 => 'Juni'
        ];

        // Baris Excel Tanggal 1 dimulai (Berdasarkan File Anda)
        $startRow = 10; 

        foreach ($months as $monthNum => $sheetName) {
            
            // Cek apakah sheet ada
            $sheet = $spreadsheet->getSheetByName($sheetName);
            if (!$sheet) continue;

            // Logika Tahun: Juli-Des = Tahun Input, Jan-Juni = Tahun Input + 1
            $currentYear = ($monthNum >= 7) ? $tahun : $tahun + 1;

            // --- QUERY DATABASE ---
            // Mengambil data sampah TERKELOLA
            $dataSampah = SampahTerkelola::whereYear('tanggal', $currentYear)
                ->whereMonth('tanggal', $monthNum)
                ->whereHas('user', function($q) use ($instansiIds) {
                    $q->whereIn('id_instansi', $instansiIds);
                })->get();

            // Jika Anda juga ingin mengambil SAMPAH DISERAHKAN (Gabung query)
            $dataDiserahkan = SampahDiserahkan::whereYear('tanggal', $currentYear)
                ->whereMonth('tanggal', $monthNum)
                ->whereHas('user', function($q) use ($instansiIds) {
                    $q->whereIn('id_instansi', $instansiIds);
                })->get();

            // Gabung Collection (Opsional: tergantung apakah mau dipisah atau digabung)
            $allData = $dataSampah->merge($dataDiserahkan);

            // --- INJECT DATA KE CELL ---
            foreach ($allData as $row) {
                
                // 1. Tentukan Baris (Row)
                // Ambil tanggal hari ini (1-31)
                $tgl = (int)Carbon::parse($row->tanggal)->format('d');
                
                // Rumus: Baris Awal + (Tanggal - 1)
                // Contoh: Tgl 1 -> 10 + 0 = Baris 10
                // Contoh: Tgl 5 -> 10 + 4 = Baris 14
                $barisExcel = $startRow + ($tgl - 1); 

                // 2. Tentukan Kolom (Column)
                // Panggil fungsi Helper di bawah
                $kolomExcel = $this->getExcelColumn($row->id_lokasi, $row->id_jenis);

                if ($kolomExcel) {
                    // Ambil nilai yang sudah ada (supaya bisa dijumlahkan jika ada double input)
                    $nilaiLama = $sheet->getCell($kolomExcel . $barisExcel)->getValue();
                    
                    // Jumlahkan nilai lama + nilai baru
                    $totalBaru = (float)$nilaiLama + (float)$row->jumlah_berat;
                    
                    // Tulis kembali ke Excel
                    $sheet->setCellValue($kolomExcel . $barisExcel, $totalBaru);
                }
            }
        }

        // ====================================================
        // BAGIAN 3: DOWNLOAD FILE
        // ====================================================
        
        $writer = new Xlsx($spreadsheet);
        $fileName = 'Laporan_Neraca_Sampah_' . $tahun . '.xlsx';
        
        return response()->streamDownload(function() use ($writer) {
            $writer->save('php://output');
        }, $fileName);
    }

    // ====================================================
    // FUNGSI HELPER: PEMETAAN KOLOM
    // ====================================================
    // Fungsi ini menentukan ID Lokasi & ID Jenis masuk ke Kolom Huruf Apa
    
    private function getExcelColumn($lokasiId, $jenisId)
    {
        // ASUMSI ID JENIS SAMPAH DARI DATABASE:
        // 1 = Organik
        // 2 = Anorganik
        // 3 = Residu / B3 / Lainnya

        // LOKASI 1: AREA KANTOR
        if ($lokasiId == 1) {
            if ($jenisId == 1) return 'C'; // Organik
            if ($jenisId == 2) return 'D'; // Anorganik
            if ($jenisId == 3) return 'E'; // Residu
        }

        // LOKASI 2: AREA PARKIR / TAMAN
        if ($lokasiId == 2) {
            if ($jenisId == 1) return 'G'; // Organik
            if ($jenisId == 2) return 'H'; // Anorganik
            if ($jenisId == 3) return 'I'; // Residu
        }

        // LOKASI 3: RUANG TUNGGU
        if ($lokasiId == 3) {
            if ($jenisId == 1) return 'K'; // Organik
            if ($jenisId == 2) return 'L'; // Anorganik
            if ($jenisId == 3) return 'M'; // Residu
        }

        // LOKASI 4: TEMPAT MAKAN (KANTIN)
        if ($lokasiId == 4) {
            if ($jenisId == 1) return 'O'; // Organik
            if ($jenisId == 2) return 'P'; // Anorganik
            if ($jenisId == 3) return 'Q'; // Residu
        }

        // LOKASI 5: SAMPAH KAPAL
        if ($lokasiId == 5) {
            if ($jenisId == 1) return 'S'; // Organik
            if ($jenisId == 2) return 'T'; // Anorganik
            if ($jenisId == 3) return 'U'; // Residu
        }

        // LOKASI 6: AREA LAINNYA
        if ($lokasiId == 6) {
            if ($jenisId == 1) return 'W'; // Organik
            if ($jenisId == 2) return 'X'; // Anorganik
            if ($jenisId == 3) return 'Y'; // Residu
        }

        return null; // Jika ID tidak dikenali
    }

3. Penjelasan Logika untuk Laporan Anda
Database Query: Kita memanggil whereYear dan whereMonth sesuai loop bulan.

Filter Instansi: Kita menggunakan whereHas('user') karena ID Instansi menempel pada User, bukan pada Sampah.

Coordinate Injection: Kita tidak membuat baris baru. Kita hanya menunjuk alamat sel (misal C10) dan menimpa isinya.

Auto-Calculation: Karena sheet Rekap di Excel Anda sudah berisi rumus (misal =SUM('Juli'!C10:C40)), maka saat kita mengisi data di sheet Juli lewat kodingan ini, sheet Rekap akan otomatis terupdate saat file dibuka. Tidak perlu koding untuk sheet rekap!

Berdasarkan file template_master.xlsx (format CSV) yang Anda unggah dan perbandingan dengan file yang terisi sebelumnya, saya sudah bisa membedah Alur Logika Perhitungan (Rumus Excel) yang bekerja di balik layar.

Karena file yang Anda unggah adalah CSV (hanya nilai, bukan rumus), saya merekonstruksi logikanya berdasarkan struktur kolom dan barisnya.

Ini adalah Peta Logika yang dijalankan oleh Excel secara otomatis setelah program Laravel Anda menyuntikkan data mentah.

1. Logic Level 1: Sheet Harian (Juli - Juni)
Ini adalah "Dapur" utama. Di sini data mentah masuk, dan Excel melakukan perhitungan dasar.

Input Data (Oleh Coding Laravel):

Mengisi sel baris 10 s.d. 40 (Tanggal 1-31).

Kolom yang diisi: Berat Organik, Anorganik, Residu per Lokasi.

Rumus Excel 1: Total Per Hari (Horizontal)

Di setiap kelompok lokasi (misal Area Kantor), ada kolom "Total Timbulan Sampah".

Logika: Organik + Anorganik + Residu (pada hari itu).

Contoh (Area Kantor): F10 = C10 + D10 + E10

Rumus Excel 2: Total Per Bulan (Vertikal)

Terletak di bagian paling bawah (Baris "Total/jenis (kg/bulan)").

Logika: SUM(Baris Tanggal 1 : Baris Tanggal 31).

Contoh: =SUM(C10:C40)

Rumus Excel 3: Konversi Kg ke Ton

Terletak di baris bawahnya lagi ("Total/jenis (ton/bulan)").

Logika: Nilai Kg / 1000.

Contoh: =C42 / 1000

2. Logic Level 2: Sheet Rekap area
Sheet ini berfungsi mengumpulkan total bulanan dari setiap sheet bulan (Juli-Juni) agar bisa dilihat dalam satu tabel besar.

Input Data: Tidak ada input manual. Semua otomatis Link Formula.

Rumus Excel: Mengambil Data Sheet Lain

Setiap sel di sini mengambil nilai dari Baris Total Konversi Ton di sheet harian.

Contoh (Sel Juli - Kantor - Organik): ='Juli'!C43 (Mengambil sel total ton organik dari sheet Juli).

Contoh (Sel Agustus - Kantor - Organik): ='Agustus'!C43

3. Logic Level 3: Sheet Rekap Sampah terkelola
Sheet ini menyederhanakan data. Tidak peduli lokasinya di mana, yang penting berapa total Organik, Anorganik, dan Residu seluruh pelabuhan per bulan.

Rumus Excel: Penjumlahan Lintas Sheet

Nilai di sini adalah penjumlahan dari Total Seluruh Area yang biasanya ada di ujung kanan sheet harian.

Atau bisa juga menjumlahkan dari Sheet Rekap area.

Logika: SUM(Total Organik Kantor + Total Organik Parkir + ... + Total Organik Area Lain).

4. Logic Level 4: Sheet Rekap Neraca Pengelolaan Sampah (FINAL)
Ini adalah "Laporan Eksekutif". Isinya adalah total setahun penuh per lokasi.

Kolom "Jumlah Timbulan Sampah (ton/tahun)"

Logika: Menjumlahkan total berat (semua jenis) dari bulan Juli s.d. Juni untuk lokasi tersebut.

Rumus: =SUM('Rekap area'!Baris_Juli:Baris_Juni) (spesifik kolom total lokasi itu).

Kolom "Jumlah Sampah Organik/Anorganik/Residu"

Logika: Sama seperti di atas, tapi spesifik per jenis sampah.

Kolom "Prosentase Sampah Terkelola/Residu" (%)

Ini rumus analisa efisiensi.

Logika: (Total Berat Jenis Tertentu / Total Timbulan Sampah) x 100.

Contoh: Jika total sampah kantor 100 ton, dan organiknya 60 ton, maka rumusnya = (60 / 100) * 100 hasilnya 60%.

Anda TIDAK PERLU memikirkan Rumus Level 2, 3, dan 4 di dalam kodingan Laravel Anda.

Tugas kodingan Anda (LaporanController.php) hanyalah memastikan:

jadi yang dimasukkan kedalam file apakah yang sampah dikelola atau sampah diserahkan?

Jawabannya adalah: KEDUANYA (Dua-duanya Dimasukkan).

Alasannya sederhana: File Excel tersebut berjudul "REKAMAN TIMBULAN SAMPAH". Dalam prinsip pengelolaan lingkungan (Neraca Limbah), "Timbulan Sampah" artinya adalah total seluruh sampah yang dihasilkan di lokasi tersebut, tidak peduli nasib akhirnya mau dikelola sendiri atau diserahkan ke orang lain.

Logika Kasusnya: Misalkan di Area Kantor pada tanggal 1 Januari:

Ada sisa makanan 5 kg (Diolah jadi kompos -> Masuk tabel sampah_terkelola).

Ada botol plastik 3 kg (Dijual ke Bank Sampah -> Masuk tabel sampah_diserahkan).

Maka, di dalam Excel sheet Januari baris tanggal 1, kolom Area Kantor, totalnya harus tertulis 8 kg. Jika Anda hanya memasukkan salah satu, laporannya jadi salah (under-report).

SUMBER DATA: SAMPAH DISERAHKAN (sampah_diserahkans)

Kita harus mengubah fungsi getExcelColumn agar menerima parameter tambahan yaitu sumber data (apakah dari tabel terkelola atau diserahkan).

Silakan ganti seluruh isi LaporanController.php dengan revisi di bawah ini. Kode ini sudah memisahkan looping antara sampah terkelola dan diserahkan agar logikanya tidak tertukar.

PHP
<?php

namespace App\Http\Controllers;

use Illuminate\Http\Request;
use PhpOffice\PhpSpreadsheet\IOFactory;
use PhpOffice\PhpSpreadsheet\Writer\Xlsx;
use App\Models\SampahTerkelola;
use App\Models\SampahDiserahkan;
use App\Models\User;
use Carbon\Carbon;

class LaporanController extends Controller
{
    public function exportFinal(Request $request)
    {
        // 1. SETTING AWAL
        $tahun = $request->input('tahun'); 
        $instansiIds = $request->input('instansi_ids'); 
        
        $templatePath = storage_path('app/public/templates/template_master.xlsx');
        if (!file_exists($templatePath)) {
            return back()->with('error', 'File template master belum diupload!');
        }
        
        $spreadsheet = IOFactory::load($templatePath);

        // ====================================================
        // BAGIAN 1: ISI JUDUL INSTANSI (HEADER)
        // ====================================================
        $namaInstansi = User::whereIn('id_instansi', $instansiIds)
                        ->join('instansis', 'users.id_instansi', '=', 'instansis.id')
                        ->distinct()
                        ->pluck('instansis.nama_instansi')
                        ->toArray();

        $judulText = "SELURUH WILAYAH OPERASIONAL";
        if (count($namaInstansi) == 1) {
            $judulText = strtoupper($namaInstansi[0]);
        } elseif (count($namaInstansi) > 1 && count($namaInstansi) <= 3) {
            $judulText = strtoupper(implode(", ", $namaInstansi));
        }

        $sheetRekap = $spreadsheet->getSheetByName('Rekap Neraca Pengelolaan Sampah');
        if($sheetRekap) {
            $sheetRekap->setCellValue('C3', "Area : " . $judulText);
        }

        // ====================================================
        // BAGIAN 2: LOOPING PENGISIAN DATA HARIAN
        // ====================================================
        $months = [
            7 => 'Juli', 8 => 'Agustus', 9 => 'September', 10 => 'Oktober', 11 => 'November', 12 => 'Desember',
            1 => 'Januari', 2 => 'Februari', 3 => 'Maret', 4 => 'April', 5 => 'Mei', 6 => 'Juni'
        ];

        $startRow = 10; // Baris Tanggal 1

        foreach ($months as $monthNum => $sheetName) {
            
            $sheet = $spreadsheet->getSheetByName($sheetName);
            if (!$sheet) continue;

            $currentYear = ($monthNum >= 7) ? $tahun : $tahun + 1;

            // ---------------------------------------------------------
            // A. PROSES SAMPAH TERKELOLA (Organik -> Org, Anorganik/Residu -> Anorg)
            // ---------------------------------------------------------
            $dataTerkelola = SampahTerkelola::whereYear('tanggal', $currentYear)
                ->whereMonth('tanggal', $monthNum)
                ->whereHas('user', function($q) use ($instansiIds) {
                    $q->whereIn('id_instansi', $instansiIds);
                })->get();

            foreach ($dataTerkelola as $row) {
                $tgl = (int)Carbon::parse($row->tanggal)->format('d');
                $barisExcel = $startRow + ($tgl - 1); 
                
                // Panggil Helper dengan status 'TERKELOLA'
                $kolomExcel = $this->getExcelColumn($row->id_lokasi, $row->id_jenis, 'terkelola');

                if ($kolomExcel) {
                    $nilaiLama = $sheet->getCell($kolomExcel . $barisExcel)->getValue();
                    $sheet->setCellValue($kolomExcel . $barisExcel, $nilaiLama + $row->jumlah_berat);
                }
            }

            // ---------------------------------------------------------
            // B. PROSES SAMPAH DISERAHKAN (SEMUA -> Sampah Lainnya/Residu)
            // ---------------------------------------------------------
            $dataDiserahkan = SampahDiserahkan::whereYear('tanggal', $currentYear)
                ->whereMonth('tanggal', $monthNum)
                ->whereHas('user', function($q) use ($instansiIds) {
                    $q->whereIn('id_instansi', $instansiIds);
                })->get();

            foreach ($dataDiserahkan as $row) {
                $tgl = (int)Carbon::parse($row->tanggal)->format('d');
                $barisExcel = $startRow + ($tgl - 1); 

                // Panggil Helper dengan status 'DISERAHKAN'
                $kolomExcel = $this->getExcelColumn($row->id_lokasi, $row->id_jenis, 'diserahkan');

                if ($kolomExcel) {
                    $nilaiLama = $sheet->getCell($kolomExcel . $barisExcel)->getValue();
                    $sheet->setCellValue($kolomExcel . $barisExcel, $nilaiLama + $row->jumlah_berat);
                }
            }
        }

        // DOWNLOAD FILE
        $writer = new Xlsx($spreadsheet);
        $fileName = 'Laporan_Neraca_Sampah_' . $tahun . '.xlsx';
        
        return response()->streamDownload(function() use ($writer) {
            $writer->save('php://output');
        }, $fileName);
    }

    // ====================================================
    // FUNGSI HELPER BARU (DENGAN LOGIKA SUMBER DATA)
    // ====================================================
    private function getExcelColumn($lokasiId, $jenisId, $sumberData)
    {
        // LOGIKA KHUSUS: 
        // 1. Jika DISERAHKAN -> Semuanya masuk kolom ke-3 (Residu/Lainnya)
        // 2. Jika TERKELOLA:
        //    - Organik (1) -> Kolom 1
        //    - Anorganik (2) -> Kolom 2
        //    - Residu (3) -> Kolom 2 (Masuk ke Anorganik sesuai request)

        $targetKolom = 0; // 1=Organik, 2=Anorganik, 3=Residu/Lainnya

        if ($sumberData == 'diserahkan') {
            $targetKolom = 3; // Semua masuk kolom Residu
        } else {
            // Status Terkelola
            if ($jenisId == 1) $targetKolom = 1; // Organik
            if ($jenisId == 2) $targetKolom = 2; // Anorganik
            if ($jenisId == 3) $targetKolom = 2; // Residu masuk Anorganik
        }

        // MAPPING KOLOM BERDASARKAN LOKASI
        if ($lokasiId == 1) { // KANTOR
            if ($targetKolom == 1) return 'C';
            if ($targetKolom == 2) return 'D';
            if ($targetKolom == 3) return 'E';
        }
        if ($lokasiId == 2) { // PARKIR
            if ($targetKolom == 1) return 'G';
            if ($targetKolom == 2) return 'H';
            if ($targetKolom == 3) return 'I';
        }
        if ($lokasiId == 3) { // RUANG TUNGGU
            if ($targetKolom == 1) return 'K';
            if ($targetKolom == 2) return 'L';
            if ($targetKolom == 3) return 'M';
        }
        if ($lokasiId == 4) { // TEMPAT MAKAN
            if ($targetKolom == 1) return 'O';
            if ($targetKolom == 2) return 'P';
            if ($targetKolom == 3) return 'Q';
        }
        if ($lokasiId == 5) { // KAPAL
            if ($targetKolom == 1) return 'S';
            if ($targetKolom == 2) return 'T';
            if ($targetKolom == 3) return 'U';
        }
        if ($lokasiId == 6) { // AREA LAIN
            if ($targetKolom == 1) return 'W';
            if ($targetKolom == 2) return 'X';
            if ($targetKolom == 3) return 'Y';
        }

        return null;
    }
}
Poin Penting Perubahan:
Dua Kali Looping: Saya memisahkan foreach ($dataTerkelola) dan foreach ($dataDiserahkan). Ini memastikan kita bisa menerapkan aturan yang berbeda untuk masing-masing tabel.

Helper Logic: Di fungsi getExcelColumn, saya menambahkan logika if ($sumberData == 'diserahkan') { $targetKolom = 3; }. Ini menjamin semua sampah diserahkan (mau itu organik atau anorganik) akan "dipaksa" masuk kolom Residu/Lainnya (Kolom E, I, M, dst).

Terkelola Residu: Logika if ($jenisId == 3) $targetKolom = 2; memastikan jika ada yang salah input "Terkelola - Residu", dia akan dialihkan ke kolom Anorganik (Kolom D, H, L, dst).
